comandos necessarios para correr o projeto

(se modificar algo no frontend html)
kubectl delete configmap frontend-html
kubectl create configmap frontend-html --from-file=index.html=./frontend/index.html

(alguma mudanca no deployment)
kubectl apply -f frontend-deployment.yaml
kubectl apply -f frontend-service.yaml

kubectl apply -f catalogservice-deployment.yaml
kubectl apply -f catalogservice-service.yaml

kubectl apply -f streamingservice-deployment.yaml
kubectl apply -f streamingservice-service.yaml

kubectl apply -f uploadservice-deployment.yaml
kubectl apply -f uploadservice-service.yaml

kubectl apply -f mongodb.yaml


(no diretorio pricipal, temos que mudar os nomes para o diretorio certo)

docker build -f streamingservice/Dockerfile -t streamingservice:v1 .
minikube image load streamingservice:v1


RUN!!

kubectl port-forward service/catalogservice 5000:5000
kubectl port-forward service/streamingservice 6000:6000
kubectl port-forward service/uploadservice 7000:7000
